Формирование feature словарей (unigrams и bigrams)¶

Создаем features для классификатора. Для этого берем наиболее часто встречающиеся слова и биграммы. После первого запуска кода ниже, проходим вручную по словарям и копируем в файл stop_words.csv слова и биграммы, которые окажутся неинформативными и вредными для генерализации (н-р имена собственные и т.п.)

from collections import Counter                                                 
import re                                                                       
                                                                                
import pandas as pd                                                             
from nltk.stem.snowball import SnowballStemmer                                  
from nltk.tag import pos_tag
from nltk.util import bigrams
from tqdm import tqdm                              
                                                                                
def make_dictionary(reqs, mode = 'uni'):                                                      
    dictionary = []                                                             
    stemmer = SnowballStemmer('russian')                                        
    for req_id, req in tqdm(reqs.iterrows()):                                         
        tagged_description = pos_tag(' '.join([req['Short_Description'],        
                                               req['Detailed_Description']]).split(),
                                     lang = 'rus')                              
        #оставляем для словаря только существительные, глаголы и частицы
        if mode == 'uni':
            filter_out_some_tags = [x[0] for x in tagged_description if x[1] == 'S' or x[1] == 'V']
            stemmed_description = [stemmer.stem(x) for x in filter_out_some_tags]
            filter_out_short_words = [x for x in stemmed_description if len(x) >= 4]    
            dictionary = dictionary + filter_out_short_words                            
        elif mode == 'bi':
            filter_out_some_tags = [x[0] for x in tagged_description if x[1] == 'S' or x[1] == 'V' or x[1] == 'PART']
            stemmed_description = [stemmer.stem(x) for x in filter_out_some_tags]
            filter_out_short_words = [x for x in stemmed_description if len(x) > 1]
            descr_bigrams = bigrams(filter_out_short_words)
            dictionary = dictionary + [' '.join([x[0], x[1]]) for x in descr_bigrams]
    
    if mode == 'uni':
        result_dict = [x for x in Counter(dictionary).most_common() if x[1] >= 10]
    elif mode == 'bi':
        result_dict = [x for x in Counter(dictionary).most_common() if x[1] >= 50]
    return result_dict
                                                                                
def save_dict(dictionary, mode = 'uni'):
    dict_pd = pd.DataFrame(dictionary, columns = ['word', 'counter'])
    stop_words = pd.read_csv('stop_words.csv')   #слова которые не нужно включать в словарь - отобраны вручную                                
    dict_pd = dict_pd[~dict_pd['word'].isin(stop_words['word'].tolist())]       
    return dict_pd.to_csv(''.join((mode, 'grams_dictionary.csv')), index = False)

def vectorize(row):                                                             
    row['Detailed_Description'] = re.sub(r'[a-zA-Z:./()0-9№;_*+=@"#!?&%-]', ' ',
                                         row['Detailed_Description']).lower()   
    row['Short_Description'] = re.sub(r'[a-zA-Z:./()0-9№;_*+=@"#!?&%-]', ' ',   
                                      row['Short_Description']).lower()         
                                                                                
    dictionary = pd.read_csv('dictionary.csv')                                  
                                                                                
    stemmer = SnowballStemmer('russian')                                        
    description = [stemmer.stem(x) for x in ' '.join([row['Short_Description'],

    features_row = dictionary['word'].isin(description)                         
    features_row.index = dictionary['word']

def vectorize_by_bigrams(row):                                                             
    row['Detailed_Description'] = re.sub(r'[a-zA-Z:./()0-9№;_*+=@"#!?&%-]', ' ',
                                         row['Detailed_Description']).lower()   
    row['Short_Description'] = re.sub(r'[a-zA-Z:./()0-9№;_*+=@"#!?&%-]', ' ',   
                                      row['Short_Description']).lower()         
                                                                                
    dictionary = pd.read_csv('bigrams_dictionary.csv')                          
                                                                                
    stemmer = SnowballStemmer('russian')                                        
    description = [stemmer.stem(x) for x in row['Detailed_Description'].split()]
    short_filtered_out = [x for x in description if len(x) > 1]                 
    descr_bigrams = bigrams(short_filtered_out)                                 
    descr_bigrams_list = [' '.join([x[0], x[1]]) for x in descr_bigrams]        
                                                                                
    features_row = dictionary['word'].isin(descr_bigrams_list)                  
    features_row.index = dictionary['word']                                     
                                                                                
    return features_row
